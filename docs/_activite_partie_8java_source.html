624  public boolean estTrameValide(String trame)
625  {
626  if(trame == null)
627  {
628  return false;
629  }
630 
631  trame = trame.trim();
632  if(!trame.startsWith(CommunicationBluetooth.ENTETE) ||
633  !trame.endsWith(CommunicationBluetooth.DELIMITATEUR_FIN))
634  {
635  return false;
636  }
637 
638  String contenu = trame.substring(1, trame.length() - 1);
639  String[] morceaux = contenu.split("/");
640
