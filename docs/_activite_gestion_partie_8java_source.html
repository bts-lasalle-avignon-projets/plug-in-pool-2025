243  private int extraireNumeroTable(String texte)
244  {
245  if (texte == null || !texte.contains("(") || !texte.contains(")"))
246  {
247  Log.e("extraireNumeroTable", "Format invalide ou null : " + texte);
248  return -1;
249  }
250 
251  int indexParenthese = texte.lastIndexOf('(');
252  if (indexParenthese <= 0)
253  {
254  Log.e("extraireNumeroTable", "Parenthèse mal placée : " + texte);
255  return -1;
256  }
257 
258  String ssid = texte.substring(0, indexParenthese).trim();
259 
260  int indexTiret = ssid.lastIndexOf('-');
261  if (indexTiret == -1 || indexTiret == ssid.length() - 1)
262  {
263  Log.e("extraireNumeroTable", "Tiret manquant ou ID vide : " + ssid);
264  return -1;
265  }
266 
267  String idStr = ssid.substring(indexTiret + 1).trim();
268 
269  try
270  {
271  return Integer.parseInt(idStr);
272  }
273  catch (NumberFormatException e)
274  {
275  Log.e("extraireNumeroTable", "ID non numérique : " + idStr);
276  return -1;
277  }
